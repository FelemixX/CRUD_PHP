create table client
(
    id         int auto_increment
        primary key,
    name       varchar(256) null,
    birth_date date         null
);

create table document
(
    id            int auto_increment
        primary key,
    number        decimal(53) null,
    creation_date date        null,
    client_ID     int         null,
    constraint document_client_id_fk
        foreign key (client_ID) references client (id)
            on update cascade on delete cascade
);

create table debt
(
    id          int auto_increment
        primary key,
    client_ID   int         null,
    document_ID int         null,
    debt        decimal(56) null,
    constraint debt_client_id_fk
        foreign key (client_ID) references client (id)
            on update cascade on delete cascade,
    constraint debt_document_id_fk
        foreign key (document_ID) references document (id)
            on update cascade on delete cascade
);

create table product
(
    id          int auto_increment
        primary key,
    document_ID int          null,
    p_name      varchar(256) null,
    quantity    int          null,
    constraint product_document_id_fk
        foreign key (document_ID) references document (id)
            on update cascade on delete cascade
);

create table documents_products
(
    id          int auto_increment
        primary key,
    document_FK int null,
    product_FK  int null,
    constraint documents_products_document_id_fk
        foreign key (document_FK) references document (id)
            on update cascade on delete cascade,
    constraint documents_products_product_id_fk
        foreign key (product_FK) references product (id)
            on update cascade on delete cascade
);

create table user
(
    id       int auto_increment
        primary key,
    login    varchar(256) null,
    password varchar(256) null,
    isAdmin  tinyint(1)   null
);
